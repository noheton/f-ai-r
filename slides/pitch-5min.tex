% ---- 2. Problem ---------------------------------------------------
\begin{frame}{The problem}
  \begin{itemize}
    \item FAIR was drafted for \textbf{data}; FAIR4RS for software; FAIR4ML for models.
    \item LLM-assisted scholarly writing produces \textbf{new artefact classes}:
          transcripts, prompts, version manifests, verification ladders,
          redaction policies, \emph{per-claim provenance maps}.
    \item Volume problem on top: detection-only is an arms race the reviewer side does not win.
    \item \textbf{Reported fabrication base rates: 18--55\,\% (Walters \& Wilder), 17--34\,\% (Magesh et al.).}
  \end{itemize}
\end{frame}

\begin{frame}{Why the (AI)\textsuperscript{2} factor matters}
  \begin{itemize}
    \item Multiplicative, not appended (\emph{4AI hei{\ss}t alles}).
    \item \textbf{Eight integrated practices} as the minimum discipline.
    \item Specialisation reduces hallucination at the seam.
    \item Naming the practice is the first step toward contesting it.
  \end{itemize}
\end{frame}

% ---- 4. Eight practices -------------------------------------------
\begin{frame}{Eight integrated practices}
  \footnotesize
  \begin{tabular}{rlp{4.4cm}}
    \toprule
    \# & Practice & Failure mode addressed \\
    \midrule
    1 & Transcript preservation             & session loss \\
    2 & Verification-status labelling (FSM) & claim--evidence drift \\
    3 & Per-claim provenance maps           & fabricated citations \\
    4 & Mirror discipline                   & placeholder dishonesty \\
    5 & Recursive meta-process documentation & methodological-tract failure \\
    6 & Base-rate-anchored AI disclosure    & perfunctory disclosure \\
    7 & Legal honesty about authorship      & inflated co-author claims \\
    8 & FAIR alignment as \textbf{precondition} & retrofit cost \\
    \bottomrule
  \end{tabular}

  \vspace{0.6em}
  \textit{The integration is the contribution.} None of the eight is novel on its own.
\end{frame}
