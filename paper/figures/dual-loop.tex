% Dual loop: AI-as-writer / AI-as-reader, separated by the manuscript
% bundle. Visualises the two-pass discipline announced in
% \S\ref{sec:intro} and formalised in \S\ref{sec:axes}: the (AI)
% factor is squared because each artefact is touched twice, once by
% an authoring agent and once by an audit agent, never in the same
% role and never in the same commit. The handback boundary in the
% middle is the seam that makes specialisation reduce hallucination.
% Loaded from paper/sections/intro.tex via \input.
\begin{figure}[!t]
\centering
\begin{tikzpicture}[
  node distance=0pt,
  every node/.style={font=\sffamily\footnotesize, align=center,
                     inner sep=4pt},
  loopL/.style={draw, line width=0.5pt, draw=dlrBlau1,
                fill=dlrBlau5, minimum width=34mm,
                minimum height=20mm, text width=32mm, align=center},
  loopR/.style={draw, line width=0.5pt, draw=dlrGruen1,
                fill=dlrGruen5, minimum width=34mm,
                minimum height=20mm, text width=32mm, align=center},
  bundle/.style={draw, line width=0.5pt, draw=dlrGrau1,
                 fill=dlrGrau5, minimum width=28mm,
                 minimum height=44mm, text width=24mm,
                 align=center, text=dlrGrau1},
  arrow/.style={-{Latex[length=1.6mm]}, draw=dlrGrau1, line width=0.5pt},
  hand/.style={-{Latex[length=1.6mm]}, draw=dlrGelb1,
               line width=0.5pt, dashed},
  cap/.style={font=\sffamily\bfseries\footnotesize, text=dlrGrau1},
  legend/.style={font=\sffamily\scriptsize, text=dlrGrau1},
]
% --- Left loop: AI as writer ---------------------------------------
\node[loopL] (writer) at (-4.4, 0)
  {\textbf{AI as writer}\\[1pt]
   drafts prose\\
   proposes claims\\
   mints identifiers};

\node[cap, anchor=south] at (writer.north) {(AI)\textsubscript{1} authoring};

% Self-loop on the writer side, indicating the iterative draft cycle.
\draw[arrow] (writer.north west)
  to[out=120, in=180, looseness=2.4]
  ([yshift=-3pt]writer.north west);

% --- Right loop: AI as reader / auditor ----------------------------
\node[loopR] (reader) at (4.4, 0)
  {\textbf{AI as reader}\\[1pt]
   verifies citations\\
   validates the graph\\
   files defects};

\node[cap, anchor=south] at (reader.north) {(AI)\textsubscript{2} audit};

\draw[arrow] (reader.north east)
  to[out=60, in=0, looseness=2.4]
  ([yshift=-3pt]reader.north east);

% --- Centre: the manuscript bundle ----------------------------------
\node[bundle] (bundle) at (0, 0)
  {\textbf{manuscript}\\[2pt]
   prose\\
   sources\\
   prompts\\
   transcripts\\
   graph};

% --- Flow: writer writes into the bundle; reader reads from it ------
\draw[arrow, draw=dlrBlau1]
  (writer.east) -- node[legend, above] {writes} (bundle.west);
\draw[arrow, draw=dlrGruen1]
  (bundle.east) -- node[legend, above] {reads}  (reader.west);

% --- Handback: reader -> writer, dashed, on the yellow accent ------
\draw[hand]
  (reader.south) to[out=-150, in=-30]
  node[legend, below, midway] {\textit{handback (defect registry)}}
  (writer.south);

% --- Seam labels ----------------------------------------------------
\node[legend, anchor=south, text=dlrGrau1]
  at ([yshift=2pt]bundle.north) {shared substrate};
\end{tikzpicture}
\caption{The dual loop that the squared notation encodes. An
authoring agent (left, blue) drafts prose and proposes claims into
the manuscript bundle; an audit agent (right, green) reads the same
bundle back and verifies it. The two roles share a substrate but
never share a commit; defects travel through a handback registry
(dashed yellow) rather than through in-place edits. The seam between
writer and reader is what \S\ref{sec:handback} calls the handback
discipline; specialisation across that seam is what reduces
hallucination at the join.}
\label{fig:dual-loop}
\end{figure}
