  human/.style={fill=dlrGruen5, draw=dlrGruen1},

\node[state, ext] (un)   {\textsc{unverified-}\\\textsc{external}};
\node[state, ext, right=of un] (need) {\textsc{needs-}\\\textsc{research}};
\node[state, right=of need] (ret)  {\textsc{lit-}\\\textsc{retrieved}};
\node[state, ai, right=of ret]  (aic)  {\textsc{ai-}\\\textsc{confirmed}};
\node[state, human, right=of aic] (read) {\textsc{lit-}\\\textsc{read}};

  };

\caption{The verification ladder as a finite-state machine. A
\texttt{fair2r:Claim} progresses left-to-right as evidence
accumulates; \textsc{ai-confirmed} is the ceiling an LLM agent can
deliver on its own (\S\ref{sec:ladder}). Retraction is the only
back-edge (dashed) and is recorded as a \texttt{prov:Invalidation}
activity in \texttt{doc/provenance.ttl}, never as a deletion. A
claim may be \emph{invoked} at its current rung or below; load-bearing
or contested claims still require \textsc{lit-read}.}
